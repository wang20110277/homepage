   1: "use client";

   4: import { Button } from "@/components/ui/button";
   5: import { Card, CardContent } from "@/components/ui/card";

  11: import type { OCRResult } from "@/types";
  12: import { toast } from "sonner";
  13: import {
  14:   BadgeCheck,
  15:   Download,

  27:   { value: "structured", label: "文档结构输出", description: "保留标题 / 列表结构" },

  29:

  32:   { value: "small", label: "小型", description: "速度与准确平衡" },
  33:   { value: "base", label: "基础", description: "通用场景" },

  35: ] as const;
  36:

  39: 
  40: const features = [
  41:   { icon: Zap, text: "极速识别响应，适合批量文件" },
  42:   { icon: FileText, text: "保留段落、列表等文档结构" },

  46:

  54: export default function OCRToolPage() {

  56:   const [previewUrl, setPreviewUrl] = useState<string | null>(null);
  57:   const [mode, setMode] = useState<OCRMode>("normal");
  58:   const [modelSize, setModelSize] = useState<ModelSize>("base");
  59:   const [result, setResult] = useState<OCRResult | null>(null);

  62:   const previewRef = useRef<string | null>(null);
  63: 
  64:   useEffect(() => {
  65:     if (previewRef.current && previewRef.current !== previewUrl) {
  66:       URL.revokeObjectURL(previewRef.current);
  67:     }
  68:     previewRef.current = previewUrl;
  69:     return () => {
  70:       if (previewRef.current) {
  71:         URL.revokeObjectURL(previewRef.current);
  72:       }
  73:       if (timerRef.current) {
  74:         clearTimeout(timerRef.current);
  75:       }
  76:     };
  77:   }, [previewUrl]);
  78: 
  79:   const handleImageChange = (file: File | null, preview: string | null) => {
  80:     setImageFile(file);
  81:     setPreviewUrl(preview);
  82:     setResult(null);
  83:   };
  84: 
  85:   const buildResult = (selectedMode: OCRMode, selectedModel: ModelSize): OCRResult => {
  86:     const base = mockOCRResults[selectedMode];
  87:     const multiplier =
  88:       selectedModel === "mini" ? 0.6 : selectedModel === "small" ? 0.8 : selectedModel === "large" ? 1.2 : 1;
  89:     return {
  90:       ...base,
  91:       id: `${base.id}-${selectedModel}`,
  92:       modelSize: selectedModel,
  93:       processTime: Math.round(base.processTime * multiplier),
  94:       text: `${base.text}\n\n—— 模式：${modeOptions.find((item) => item.value === selectedMode)?.label} · 模型：${
  95:         modelOptions.find((item) => item.value === selectedModel)?.label
  96:       }`,
  97:     };
  98:   };
  99: 
 100:   const handleProcess = () => {
 101:     if (!imageFile) {
 102:       toast.error("请先上传需要识别的图片");
 103:       return;
 104:     }
 105:     setIsProcessing(true);
 106:     setResult(null);
 107: 
 108:     if (timerRef.current) {
 109:       clearTimeout(timerRef.current);
 110:     }
 111: 
 112:     timerRef.current = setTimeout(() => {

 115:       setIsProcessing(false);
 116:       toast.success("OCR 识别完成");
 117:     }, 3000);
 118:   };
 119:

 122:     try {
 123:       await navigator.clipboard.writeText(result.text);
 124:       toast.info("识别结果已复制到剪贴板");

 127:     }
 128:   };
 129: 
 130:   return (

 134:         <p className="mt-3 text-muted-foreground">
 135:           基于深度学习的智能图像文字识别，支持多种文档格式，保留结构更懂语境
 136:         </p>
 137:       </header>
 138: 
 139:       <main className="flex-1 overflow-hidden px-6 py-4">
 140:         <div className="mx-auto flex h-full max-w-6xl flex-col gap-4 overflow-hidden">
 141:           <div className="grid shrink-0 grid-cols-2 gap-4">
 142:             <Card>
 143:               <CardContent className="p-5">
 144:                 <div className="mb-3 flex items-center justify-between">
 145:                   <h2 className="text-lg font-semibold">功能亮点</h2>
 146:                   <Sparkles className="h-5 w-5 text-primary" />
 147:                 </div>
 148:                 <ul className="space-y-3 text-sm text-muted-foreground">
 149:                   {features.map((feature, index) => (
 150:                     <li key={feature.text} className="flex items-start gap-2">
 151:                       <feature.icon className="h-4 w-4 flex-shrink-0 text-primary" />
 152:                       <span>
 153:                         {index + 1}. {feature.text}
 154:                       </span>
 155:                     </li>
 156:                   ))}
 157:                 </ul>
 158:               </CardContent>
 159:             </Card>
 160: 
 161:             <Card>
 162:               <CardContent className="p-5">
 163:                 <div className="mb-3 flex items-center justify-between">
 164:                   <h2 className="text-lg font-semibold">使用提示</h2>
 165:                   <CheckCircle2 className="h-5 w-5 text-primary" />
 166:                 </div>
 167:                 <ul className="space-y-3 text-sm text-muted-foreground">
 168:                   {usageTips.map((tip, index) => (
 169:                     <li key={tip} className="flex items-start gap-2">
 170:                       <CheckCircle2 className="h-4 w-4 flex-shrink-0 text-primary" />
 171:                       <span>
 172:                         {index + 1}. {tip}
 173:                       </span>
 174:                     </li>
 175:                   ))}
 176:                 </ul>
 177:               </CardContent>
 178:             </Card>
 179:           </div>
 180: 
 181:           <div className="grid flex-1 min-h-0 grid-cols-[1.05fr,0.95fr] grid-rows-[260px_minmax(0,1fr)] gap-4">
 182:             <Card className="row-span-1">
 183:               <CardContent className="flex h-full flex-col p-4">
 184:                 <Label className="mb-3 text-sm font-medium">上传图像</Label>

 186:                   <ImageUpload file={imageFile} previewUrl={previewUrl} onChange={handleImageChange} compact />
 187:                 </div>
 188:               </CardContent>
